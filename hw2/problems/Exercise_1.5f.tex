A DFA that accepts the language \setsuch{w}{w \notin \text{a*} \cup \text{b*}} must be constructed.
\par
Let $M$ be the DFA that accepts any string in the language a* $\cup$ b*:
\begin{figure}[H] \centering \caption*{$M$} \begin{tikzpicture}
	\GraphInit[vstyle=Dijkstra]
	\tikzset{LabelStyle/.style = {fill=white}}
	
	\Vertex[Math, x=0, y=2]{q_0}
	\Vertex[Math, x=2, y=2]{q_1}
	\Vertex[Math, x=0, y=0]{q_2}
	\Vertex[Math, x=2, y=0]{q_3}
	
	\Edge[label=b,style={->, bend left}](q_0)(q_1)
	\Edge[label=a,style={->, bend left}](q_0)(q_2)
	\Edge[label=a,style={->, bend left}](q_1)(q_3)
	\Edge[label=b,style={->, bend left}](q_2)(q_3)
	
	\Loop[label=a,dist=1cm,dir=WE,style{->}](q_2)
	\Loop[label=b,dist=1cm,dir=EA,style{->}](q_1)
	\Loop[label={a,b},dist=1cm,dir=SOEA,style{->}](q_3)
	
	\draw (0,2) circle (0.25cm);
	\draw (2,2) circle (0.25cm);
	\draw (0,0) circle (0.25cm);
	\draw[thick,->] (0,2.75) -- (0,2.3);
	
\end{tikzpicture} \end{figure}
